<script>
    let entriesData = [];
    window.onload = function() { updateTickerDate(); };

    function updateTickerDate() {
        const today = new Date();
        document.getElementById('tickerLiveDate').innerText = `${String(today.getDate()).padStart(2, '0')}/${String(today.getMonth() + 1).padStart(2, '0')}/${today.getFullYear()}`;
    }

    function getTodayDateString() {
        const today = new Date();
        return `${String(today.getDate()).padStart(2, '0')}/${String(today.getMonth() + 1).padStart(2, '0')}/${today.getFullYear()}`;
    }

    function formatTo12Hour(dateObj) {
        let hours = dateObj.getHours();
        let minutes = dateObj.getMinutes();
        const ampm = hours >= 12 ? 'PM' : 'AM';
        hours = hours % 12 || 12;
        return hours + ':' + (minutes < 10 ? '0' + minutes : minutes) + ' ' + ampm;
    }

    function resizeAndCompressImage(file, callback) {
        if (!file) { callback(""); return; }
        const reader = new FileReader();
        reader.onload = function(e) {
            const img = new Image();
            img.onload = function() {
                const canvas = document.createElement('canvas');
                let w = img.width, h = img.height;
                if (w > h) { if (w > 120) { h *= 120 / w; w = 120; } } 
                else { if (h > 120) { w *= 120 / h; h = 120; } }
                canvas.width = w; canvas.height = h;
                canvas.getContext('2d').drawImage(img, 0, 0, w, h);
                callback(canvas.toDataURL('image/jpeg', 0.6));
            };
            img.src = e.target.result;
        };
        reader.readAsDataURL(file);
    }

    function recalculateLiveTickerStats() {
        let tv = 0, tp = 0, tm = 0, tf = 0, tc = 0, cars = 0, bikes = 0, scooties = 0, others = 0;
        entriesData.forEach(item => {
            tp += item.totalPassengers; tm += item.males; tf += item.females; tc += item.children;
            if(item.vehicleType || item.vehicleNo) tv++;
            const t = item.vehicleType.toLowerCase();
            if (t === 'car') cars++; else if (t === 'bike') bikes++; else if (t === 'scooty') scooties++; else if (t === 'others') others++;
        });
        document.getElementById('statVehicles').innerText = tv;
        document.getElementById('statPassengers').innerText = tp;
        document.getElementById('statCars').innerText = cars;
        document.getElementById('statBikes').innerText = bikes;
        document.getElementById('statScooties').innerText = scooties;
        document.getElementById('statOthers').innerText = others;
        document.getElementById('statMales').innerText = tm;
        document.getElementById('statFemales').innerText = tf;
        document.getElementById('statChildren').innerText = tc;
    }

    function processAndAddEntry() {
        resizeAndCompressImage(document.getElementById('photo').files[0], function(imgData) {
            const m = parseInt(document.getElementById('passMales').value) || 0;
            const f = parseInt(document.getElementById('passFemales').value) || 0;
            const c = parseInt(document.getElementById('passChildren').value) || 0;
            entriesData.push({
                id: Date.now(), date: getTodayDateString(), entryTime: formatTo12Hour(new Date()), exitTime: "Still In",
                name: document.getElementById('name').value, mobile: document.getElementById('mobile').value,
                idType: document.getElementById('idType').value, idNumber: document.getElementById('idNumber').value,
                vehicleType: document.getElementById('vehicleType').value || "", vehicleNo: document.getElementById('vehicleNo').value || "",
                males: m, females: f, children: c, totalPassengers: (m + f + c),
                address: document.getElementById('address').value || "N/A", city: document.getElementById('city').value || "N/A",
                state: document.getElementById('state').value, driverName: document.getElementById('driverName').value || "N/A",
                driverId: document.getElementById('driverId').value || "N/A", emergencyContact: document.getElementById('emergencyContact').value,
                remarks: document.getElementById('remarks').value || "N/A", photoData: imgData
            });
            renderTableData(); recalculateLiveTickerStats(); document.getElementById('entryForm').reset();
        });
    }

    function renderTableData() {
        const tbody = document.querySelector('#logTable tbody'); tbody.innerHTML = "";
        entriesData.forEach(item => {
            const row = document.createElement('tr');
            row.innerHTML = `
                <td>${item.date}</td><td><span class="time-badge">${item.entryTime}</span></td>
                <td>${item.exitTime === "Still In" ? `<span class="time-badge exit-badge">Still In</span><br><button type="button" class="btn-exit-trigger" onclick="checkoutVisitor(${item.id})">Stamp Exit</button>` : `<span class="time-badge">${item.exitTime}</span>`}</td>
                <td><strong>${item.name}</strong></td><td>${item.mobile}</td><td><span class="demo-pill">${item.idType}</span> <code>${item.idNumber}</code></td>
                <td>${item.vehicleType ? `<strong>${item.vehicleType}</strong><br><small>${item.vehicleNo.toUpperCase()}</small>` : 'N/A'}</td>
                <td>${item.address}, ${item.city}, <br><small><strong>${item.state}</strong></small></td>
                <td><span class="demo-pill">M: ${item.males}</span><span class="demo-pill">W: ${item.females}</span><span class="demo-pill">C: ${item.children}</span></td>
                <td style="text-align: center; font-weight: 800;">${item.totalPassengers}</td>
                <td>${item.driverName !== 'N/A' ? `Name: ${item.driverName}<br><small>ID: ${item.driverId}</small>` : 'N/A'}</td>
                <td>${item.emergencyContact}</td><td><small>${item.remarks}</small></td>
                <td>${item.photoData ? `<img src="${item.photoData}" class="img-preview">` : `<span class="no-img-placeholder">No Photo</span>`}</td>
            `;
            tbody.appendChild(row);
        });
        document.getElementById('count').innerText = entriesData.length;
    }

    function checkoutVisitor(id) {
        const idx = entriesData.findIndex(e => e.id === id);
        if (idx !== -1) { entriesData[idx].exitTime = formatTo12Hour(new Date()); renderTableData(); }
    }

    function filterTable() {
        const q = document.getElementById('tableSearch').value.toLowerCase();
        document.querySelectorAll('#logTable tbody tr').forEach(r => {
            r.style.display = r.textContent.toLowerCase().includes(q) ? "" : "none";
        });
    }

    function exportToExcel() {
        if (!entriesData.length) { alert("Registry empty."); return; }
        const worksheet = XLSX.utils.json_to_sheet(entriesData.map((item, idx) => ({
            "Serial No": idx + 1, "Entry Date": item.date, "Entry Time": item.entryTime, "Exit Time": item.exitTime,
            "Visitor Name": item.name, "Mobile Number": item.mobile, "ID Type": item.idType, "ID Number": item.idNumber,
            "Vehicle Type": item.vehicleType || "N/A", "Vehicle Plate": item.vehicleNo || "N/A", "Men": item.males, "Women": item.females,
            "Children": item.children, "Total Passengers": item.totalPassengers, "Address": item.address, "City": item.city,
            "State": item.state, "Driver Name": item.driverName, "Driver ID": item.driverId, "Emergency Contact": item.emergencyContact, "Remarks": item.remarks
        })));
        const workbook = XLSX.utils.book_new(); XLSX.utils.book_append_sheet(workbook, worksheet, "Registry");
        XLSX.writeFile(workbook, `TARA_Registry_${getTodayDateString().replace(/\//g, "-")}.xlsx`);
    }
